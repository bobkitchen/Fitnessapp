import Foundation

/// Centralized storage for API credentials
/// These values are used as fallbacks if Keychain values are not set
enum AppSecrets {

    // MARK: - Strava API

    static let stravaClientId = "YOUR_STRAVA_CLIENT_ID"
    static let stravaClientSecret = "YOUR_STRAVA_CLIENT_SECRET"

    // MARK: - Helper Methods

    /// Initialize Keychain with secrets on first launch
    static func initializeKeychainIfNeeded() {
        if !KeychainService.hasStravaCredentials {
            try? KeychainService.saveStravaCredentials(
                clientId: stravaClientId,
                clientSecret: stravaClientSecret
            )
        }
    }
}
